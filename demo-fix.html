StickyContainer Demo - Правильное использование

Проблема в вашем коде:

1. Селектор .modal-container ищет элемент, но StickyContainer находится внутри него

2. React компоненты лучше работают с ref, а не с селекторами

3. Нужно обеспечить, что контейнер уже смонтирован когда StickyContainer пытается его найти

// ❌ Неправильно
<div className="modal-container">
  <StickyContainer container=".modal-container"> {/* Ищет сам себя! */}
    ...
  </StickyContainer>
</div>

// ✅ Правильно - вариант 1 (с useEffect)
const containerRef = useRef<HTMLDivElement>(null);
const [container, setContainer] = useState<HTMLElement | null>(null);

useEffect(() => {
  setContainer(containerRef.current);
}, []);

<div ref={containerRef}>
  {container && (
    <StickyContainer container={container}>
      ...
    </StickyContainer>
  )}
</div>

// ✅ Правильно - вариант 2 (с callback ref)
const [container, setContainer] = useState<HTMLElement | null>(null);

<div ref={setContainer}>
  {container && (
    <StickyContainer container={container}>
      ...
    </StickyContainer>
  )}
</div>
